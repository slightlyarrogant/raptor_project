Parametry Pracy - własne ustawienia
Poniżej przykład dodawania własnych ustawień w pluginach.
using Amag.Services.Settings.Interfaces;

[Export(typeof(ICustomSettingsProvider))]
class MyPluginSettings : ICustomSettingsProvider
{
    public void Fill(ISettingsContextModel model)
    {
        var rootGroup = model.GetOrAddGroup("Mój Plugin");
        var childGroup = rootGroup.GetOrAddGroup("CRM");

        childGroup.AddLink("Test Link", "vendo://crm/klienci/karta?id=2");
        childGroup.AddPercent("Test Percent", "nazwa_w_tc_config_1");
        childGroup.AddBoolean("Test Bool", "nazwa_w_tc_config_2");

        var options = new[] { 
            new SettingsEntryOptionModel("Value 1", "1"), 
            new SettingsEntryOptionModel("Value 2", "2") });
        childGroup.AddComboBox("Test Combo","nazwa_w_tc_config_3", options);
    }
}
